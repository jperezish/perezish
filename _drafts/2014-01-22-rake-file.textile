h1. Rake File

Using rake for routine tasks.

{% highlight ruby %}
  require 'rspec/core/rake_task'
  
  desc 'Default: run specs.'
  task :default => :spec
  
  desc "Run specs"
  RSpec::Core::RakeTask.new do |t|
    t.pattern = #"name_of_your_spec.rb"
    t.rspec_opts = "--color -f progress"
  end
{% endhighlight %}
